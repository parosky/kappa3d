 18     /**
 19      * @name makeRotationMatrix
 20      * @description 任意の軸まわりの3x3回転行列を作成する
 21      * @param {nector} axis 回転軸を表す単位ベクトル
 22      * @param {Vector} radian 回転角
 23      * @returns {Matrix} 作成された変換行列
 24      */
 25     function makeRotationMatrix(axis, radian) {
 26         var R = new Matrix(3, 3, false);
 27         var R1 = new Matrix(3, 3, false);
 28         var R2 = new Matrix(3, 3, true);
 29         var R3 = new Matrix(3, 3, false);
 30 
 31         R1.e[0][0] = axis.e[0] * axis.e[0];
 32         R1.e[0][1] = axis.e[0] * axis.e[1];
 33         R1.e[0][2] = axis.e[0] * axis.e[2];
 34         R1.e[1][0] = axis.e[1] * axis.e[0];
 35         R1.e[1][1] = axis.e[1] * axis.e[1];
 36         R1.e[1][2] = axis.e[1] * axis.e[2];
 37         R1.e[2][0] = axis.e[2] * axis.e[0];
 38         R1.e[2][1] = axis.e[2] * axis.e[1];
 39         R1.e[2][2] = axis.e[2] * axis.e[2];
 40 
 41         R3.e[0][1] = -axis.e[2];
 42         R3.e[0][2] = axis.e[1];
 43         R3.e[1][0] = axis.e[2];
 44         R3.e[1][2] = -axis.e[0];
 45         R3.e[2][0] = -axis.e[1];
 46         R3.e[2][1] = axis.e[0];
 47 
 48         // R = R1 + (R2 - R1) * cos(radian) - R3 * sin(radian)
 49         R = R1;
 50         R = R.add(R2.sub(R1).x(Math.cos(radian)));
 51         R = R.sub(R3.x(Math.sin(radian)));
 52 
 53         return R;
 54     }

 56     /**
 57      * @name makeAffineMatrix
 58      * @description ふたつの三角形を対応させるアフィン変換行列を求める
 59      * @param {Vector} [V1, V2, V3] 変換元の三角形の各頂点
 60      * @param {Vector} [V4, V5, V6] 変換先の三角形の各頂点
 61      * @return {Matrix} 計算された変換行列
 62      */
 63     function makeAffineMatrix(V1, V2, V3, V4, V5, V6) {
 64         var m = new Matrix(3, 3, false);
 65 
 66         // translation
 67         m.e[0][2] = V4.sub(V1).e[0];
 68         m.e[1][2] = V4.sub(V1).e[1];
 69         m.e[2][2] = 1;
 70 
 71         // rotation
 72         var x1 = V2.sub(V1).e[0];
 73         var x2 = V3.sub(V1).e[0];
 74         var y1 = V2.sub(V1).e[1];
 75         var y2 = V3.sub(V1).e[1];
 76         var x1d = V5.sub(V4).e[0];
 77         var x2d = V6.sub(V4).e[0];
 78         var y1d = V5.sub(V4).e[1];
 79         var y2d = V6.sub(V4).e[1];
 80         m.e[0][0] = (x1d * y2 - x2d * y1) / (x1 * y2 - x2 * y1);
 81         m.e[0][1] = (x1 * x2d - x1d * x2) / (x1 * y2 - x2 * y1);
 82         m.e[1][0] = (y1d * y2 - y1 * y2d) / (x1 * y2 - x2 * y1);
 83         m.e[1][1] = (x1 * y2d - x2 * y1d) / (x1 * y2 - x2 * y1);
 84 
 85         return m;
 86     }

338     /**
339      * @name Matrix
340      * @class Kappa3D のための行列クラス
341      * @property {Object} e 行列の各要素を格納する2次元配列
342      * @description 新しい行列を作成する
343      * @param {Number} row 行数
344      * @param {Number} col 列数
345      * @param {Boolean} identity true ならば単位行列、false ならばゼロ行列
346      * @return {Matrix} 作成された行列
347      */
348     function Matrix(row, col, identity) {
349         this.e = [];
350         for (var i = 0; i < row; i++) {
351             // new column
352             this.e[i] = [];
353             for (var j = 0; j < col; j++) {
354                 if (identity && i === j) {
355                     // identity vector
356                     this.e[i][j] = 1;
357                 } else {
358                     // zero vector
359                     this.e[i][j] = 0;
360                 }
361             }
362         }
363     }

366      * @description 行列を加算する
367      * @param {Matrix} matrix 加算対象となる行列
368      * @return {Matrix} 加算後の行列
369      */
370     Matrix.prototype.add = function (t) {
371         var ret = clone(this);
372         var rows, cols;
373         rows = this.e.length;
374         cols = this.e[0].length;

377             for (var col = 0; col < cols; col++) {
378                 ret.e[row][col] += t.e[row][col];

422     /**
423      * @description 行列とベクトルの掛け算を行う
424      * @param {Object} t 掛け合わせる Vector
425      * @return {Object} 計算後の Vector
426      */
427     Matrix.prototype.dot_v = function (t) {
428         var ret, rows, cols, sum;
429 
430         // vector
431         ret = [];
432         rows = this.e.length;
433         var _cols = this.e[0].length;
434         var e = this.e;
435         var te = t.e;
436         while (rows--) {
437             sum = 0;
438             cols = _cols;
439             while (cols--) {
440                 sum += e[rows][cols] * te[cols];
441             }
442             ret[rows] = sum;
443         }
444         return $V(ret);
445     };

447     /**
448      * @description 行列と行列の掛け算を行う
449      * @param {Object} t 掛け合わせる Matrix
450      * @return {Object} 計算後の Matrix
451      */
452     Matrix.prototype.dot_m = function (t) {
453         var ret, rows, cols, sum;
454 
455         ret = new Matrix(this.e.length, t.e[0].length, false);
456         var k;
457         var kk = this.e[0].length;
458         rows = this.e.length;
459         cols = this.e[0].length;
460 
461         var row, col;
462         var e = this.e;
463         var te = t.e;
464         row = rows;
465         while (row--) {
466             col = cols;
467             while (col--) {
468                 k = kk;
469                 sum = 0;
470                 while (k--) {
471                     sum += e[row][k] * te[k][col];
472                 }
473                 ret.e[row][col] = sum;
474             }
475         }
476         return ret;
477     };

544     /**
545      * @description カメラ情報をもとにビュー変換行列を作成する
546      */
547     Scene.prototype.setViewMatrix = function () {
548         var u = this.camera.upper;
549         var z = this.camera.direction.x(-1);
550         var x = u.cross(z);
551         var y = z.cross(x);
552         var e = this.camera.center;
553 
554         var viewM = new Matrix(4, 4, false);
555 
556         for (var i = 0; i < 3; i++) {
557             viewM.e[0][i] = x.e[i];
558             viewM.e[1][i] = y.e[i];
559             viewM.e[2][i] = z.e[i];
560         }
561         viewM.e[0][3] = -e.dot(x);
562         viewM.e[1][3] = -e.dot(y);
563         viewM.e[2][3] = -e.dot(z);
564         viewM.e[3][3] = 1;
565 
566         this.viewM = viewM;
567 
568         // compute persM * viwM for speed up
569         this.persviewM = this.persM.dot_m(this.viewM);
570     };

572     /**
573      * @description 透視投影変換行列を作成する
574      * @param {Number} n ニアクリップ面への距離
575      * @param {Number} f ファークリップ面への距離
576      * @param {Number} fov 視野角の大きさ
577      */
578     Scene.prototype.setPerspectiveMatrix = function (n, f, fov) {
579         var w = 2 * Math.tan(fov / 2);
580         var h = w * this.aspect;
581         var l = -w / 2;
582         var r = w / 2;
583         var t = h / 2;
584         var b = -h / 2;
585 
586         var persM = new Matrix(4, 4, false);
587         persM.e[0][0] = (2 * n) / (r - l);
588         persM.e[0][2] = (r + l) / (r - l);
589         persM.e[1][1] = (2 * n) / (t - b);
590         persM.e[1][2] = (t + b) / (t - b);
591         persM.e[2][2] = -(f + n) / (f - n);
592         persM.e[2][3] = -2 * n * f / (f - n);
593         persM.e[3][2] = -1;
594 
595         this.persM = persM;
596     };

851     /**
852      * @description シーン上にポリゴンを描画する
853      */
854     Polygon.prototype.draw = function () {
855         var i, ii;
856 
857         // check backside
858         var angle = this.scene.camera.direction.angleFrom(this.normal());
859         if (!this.show_back && angle < 90 * Math.PI / 180) {
860             // if backside of the polygon is seen
861             return;
862         }
863 
864         // compute screen coordinate of each vertex
865         var v = this.v;
866         var points = [];
867         ii = v.length;
868         for (i = 0; i < ii; i++) {
869             // view and perspective translation
870             points[i] = this.scene.persviewM.dot_v(v[i].to4D());
871 
872             // normalization
873             for (var j = 0; j < 4; j++) {
874                 points[i].e[j] /= points[i].e[3];
875             }
876 
877             // screen translation
878             points[i] = this.scene.screenM.dot_v(points[i]);
879 
880             // to int
881             points[i] = points[i].toInt();
882         }
883 
884         // draw
885         var c;
886         var normal;
887         var radian;
888         var ctx = this.scene.ctx;
889         if (!this.texture) {
890             // no texture
891 
892             // make path
893             ctx.beginPath();
894             ctx.moveTo(points[0].e[0], points[0].e[1]);
895             ii = points.length;
896             for (i = 1; i < ii; i++) {
897                 ctx.lineTo(points[i].e[0], points[i].e[1]);
898             }
899             ctx.closePath();
900 
901             // compute color
902             // color = light.ambient * this.color + light.diffuse * diffuse
903             // diffuse = this.color .* light.color * abs(cos(radian))
904             normal = this.normal();
905             radian = normal.angleFrom(this.scene.light.direction);
906             var diffuse = this.color.x_each(this.scene.light.color).x(Math.abs(Math.cos(radian)));
907             c = this.color.x(255).x(this.scene.light.ambient).add(diffuse.x(this.scene.light.diffuse));
908             ctx.fillStyle = "rgb(" + Math.floor(c.e[0]) + ", " + Math.floor(c.e[1]) + ", " + Math.floor(c.e[2]) + ")";
909 
910             // draw
911             ctx.fill();
912         } else {
913             // with texture
914 
915             // lighting color
916             // color = this.color .* scene.color * abs(cos(radian))
917             // color.a = 1 - abs(cos(radian))
918             if (!this.no_light) {
919                 normal = this.normal();
920                 radian = normal.angleFrom(this.scene.light.direction);
921                 c = this.color.x_each(this.scene.light.color).x(Math.abs(Math.cos(radian)));
922                 ctx.fillStyle = "rgba(" + Math.floor(c.e[0]) + ", " + Math.floor(c.e[1]) + ", " + Math.floor(c.e[2]) + ", " + (1 - Math.abs(Math.cos(radian)) / 1.2) + ")";
923             }
924 
925             // draw par triangle
926             ii = points.length - 2;
927             for (i = 0; i < ii; i++) {
928                 // clipping
929                 ctx.save();
930                 ctx.beginPath();
931                 ctx.moveTo(points[0].e[0], points[0].e[1]);
932                 ctx.lineTo(points[i + 1].e[0], points[i + 1].e[1]);
933                 ctx.lineTo(points[i + 2].e[0], points[i + 2].e[1]);
934                 ctx.closePath();
935                 ctx.clip();
936 
937                 // draw texture
938                 var m = makeAffineMatrix(this.texture.v[0], this.texture.v[i + 1], this.texture.v[i + 2], points[0], points[i + 1], points[i + 2]);
939                 ctx.setTransform(m.e[0][0], m.e[1][0], m.e[0][1], m.e[1][1], m.e[0][2], m.e[1][2]);
940                 ctx.drawImage(this.texture.image, 0, 0);
941 
942                 // lighting
943                 if (!this.no_light) {
944                     ctx.fill();
945                 }
946                 ctx.restore();
947             }
948         }

1276     /**
1277      * @description 動作パターンを実行する。メインループ内で毎フレームごとに呼び出される。
1278      */
1279     K3DObject.prototype.update = function () {
1280         if (!this.animation) {
1281             return;
1282         }
1283 
1284         var anim = this.animation;
1285         var i, ii;
1286 
1287         // copy anim.pattern to anim.queue if queue is empty 
1288         if (anim.queue.length === 0 && anim.repeat) {
1289             ii = anim.pattern.length;
1290             for (i = 0; i < ii; i++) {
1291                 anim.queue.push(clone(anim.pattern[i]));
1292             }
1293         }
1294 
1295         // animation
1296         if (anim.queue.length !== 0) {
1297             var q = anim.queue[0];
1298 
1299             // q[3] = q[2] - passed time
1300             if (!q[3]) {
1301                 q[3] = q[2];
1302             }
1303 
1304             // next task if task was done
1305             if (q[3] <= 0) {
1306                 anim.queue.shift();
1307                 return;
1308             }
1309 
1310             // msec for this frame
1311             var msec = 1000 / this.scene.fps;
1312             q[3] -= msec;
1313 
1314             var p;
1315             switch (q[0]) {
1316             case "scale":
1317                 var e = [];
1318                 ii = q[1].e.length;
1319                 for (i = 0; i < ii; i++) {
1320                     e[i] = Math.pow(q[1].e[i], msec / q[2]);
1321                 }
1322                 this.scale($V(e));
1323                 break;
1324             case "moveUp":
1325                 p = 1.0 * q[1] * (1 / q[2] * msec);
1326                 this.moveUp(p);
1327                 break;
1328             case "moveLeft":
1329                 p = 1.0 * q[1] * (1 / q[2] * msec);
1330                 this.moveLeft(p);
1331                 break;
1332             case "moveForward":
1333                 p = 1.0 * q[1] * (1 / q[2] * msec);
1334                 this.moveForward(p);
1335                 break;
1336             case "rotateYaw":
1337                 p = 1.0 * q[1] * (1 / q[2] * msec);
1338                 this.rotateYaw(p);
1339                 break;
1340             case "rotateRoll":
1341                 p = 1.0 * q[1] * (1 / q[2] * msec);
1342                 this.rotateRoll(p);
1343                 break;
1344             case "rotatePitch":
1345                 p = 1.0 * q[1] * (1 / q[2] * msec);
1346                 this.rotatePitch(p);
1347                 break;
1348             case "look":
1349                 p = q[1];
1350                 this.look(p);
1351                 break;
1352             }
1353         }
